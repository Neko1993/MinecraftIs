- Возьми зонтик! На улице дождь!

```python
weather = [RAIN, CLEAR, THUNDER]

def get_weather():
    global weather
    w = randint(0, 2)
    return weather[w]

def on_chat():
    w = get_weather()
    if w == RAIN:
        player.say("Возьми зонтик! На улице дождь!")
    elif w == CLEAR:
        player.say("На улице тепло и солнечно!")
    elif w == THUNDER:
        player.say("Лучше остаться дома")
player.on_chat('Что_одеть', on_chat)
```
